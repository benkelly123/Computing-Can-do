Day of the Week

The algorithm tells you the day of the week given a specific date.

The algorithm works by having March as the first month. It also works in centurys, so 2012 will be the 20th century.

# This program works out the day of the week from a given date
def main():
days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"]
print("Days of the week")
again = 'y'
while again != 'n':
date = split_date(get_date())
while len(date) != 3:
print("Please Try again")

def get_date():
# Simple input function, returns a string
date_string = input("Date (dd/mm/yyyy): ")
return date_string
def split_date(date_string):
# Takes a string and tries to make it into a list
try:
date = date_string.split('/')
for i in range(3):
date[i] = int(date[i])
return date
except (IndexError, ValueError):
return []
def convert(date):
# Takes the date as a list and computes the day of the week
day = date[0]
(century, year) = divmod(date[2], 100)
# In the formula March is month 1
month = ((date[1] - 3) % 12) + 1
# January and feburary are therefore in the previous year!
if month == 11 or month == 12:
year -= 1
a = int(2.6 * month - 0.2)
b = year // 4
c = century // 4
d = 2 * century
return (day + a + year + b + c - d)
# main
main()

Minimum and Maximum

This uses absolute values to calculate the minimum and maximum values.

# Minimum and maximum
def my_min(x, y):
# Takes two numbers and returns the smallest
return (x + y - abs(a - y)) / 2
def my_max(x, y):
# Takes two numbers and returns the largest
return (x + y + abs(a - y)) / 2
#main
if __name__ == "__main__":
print("Minimum and maximum numbers")
a = float(input("a: "))
b = float(input("b: "))
print("max: ", my_max(a, b))
print("min: ", my_min(a, b))

Averages

Here is an example for obtaining the average from a list or a number of arguments.

# Average number example
def average(x, *args):
#Takes a list or a series of arguments and returns the value
if type(x) == list:
total = 0
for n in x:
total +=n
return total/len(x)
else:
for n in args:
